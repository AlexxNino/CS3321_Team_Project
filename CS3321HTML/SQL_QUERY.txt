DROP DATABASE school;

CREATE DATABASE school;

CREATE TABLE courses (
    Course_name VARCHAR(25) ,
    Course_Number INT(4),
    Teacher_Name VARCHAR(25)
    );

CREATE TABLE grades (
    course_name VARCHAR(25),
    Course_Number INT(4),
    student_id INT(5),
    grade INT(3),
    Semester VARCHAR(10)
);

CREATE TABLE student (
    fname varchar(25),
    lname varchar(25),
    student_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    email varchar(45)
    );
